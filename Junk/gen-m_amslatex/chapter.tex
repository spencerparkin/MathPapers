\includeonly{preface,chap1,biblio,index}

\theoremstyle{definition}

\newtheorem{example}[theorem]{Example}
\newtheorem{xca}[theorem]{Exercise}

\theoremstyle{remark}

\numberwithin{equation}{chapter}
